📋 Codice Modificato

updateChipsDisplay() - PRIMA (BUG)

const currentSelections = state.selections[state.currentTab] || [];
// Mostrava solo selezioni del tab corrente

updateChipsDisplay() - DOPO (FIX)

// Raccoglie TUTTE le selezioni da TUTTE le categorie
const allSelections = [];
Object.keys(state.selections).forEach(category => {
    state.selections[category].forEach(item => {
        allSelections.push({
            ...item,
            category: category
        });
    });
});
// Mostra tutti gli elementi selezionati
